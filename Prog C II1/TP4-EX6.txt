#include<stdio.h>

void PERMUTER(int *a, int *b) {
    int aux;
    aux = *a;
    *a = *b;
    *b = aux;
}

int TRANSPO_MAT(int MAT[][10], int L, int LMAX, int C, int CMAX) {
    int i, j;
    if ((L < LMAX) && (C < CMAX)) {
        for (i = 0; i < L; i++) {
            for (j = i + 1; j < C; j++) {
                PERMUTER(&MAT[i][j], &MAT[j][i]);
            }
        }
        return 1;
    } else {
        return 0;
    }
}
